<!-- handleServerMessage dispatch (TerminalPane.tsx lines 102-220) — the extraction target -->
The 16-case switch dispatches to these callbacks:
- output -> writeRef.current(data)
- pty-ready -> setCurrentShell(resolvedName)
- pty-exit -> writeRef(exit msg), recordCompleted(), notify(), auto-respawn
- shell-list -> setShells(shells)
- error -> writeRef(error msg)
- save-image-result -> setPendingImagePath(path)
- window-thumbnails -> setPickerWindows(windows)
- capture-result-with-metadata -> setPendingInjection(formatCaptureBlock(...))
- agent-event -> useAgentEventStore.getState().pushEvent(event)
- pm-chat-token -> usePmChatStore.getState().appendToken(requestId, token)
- pm-chat-done -> usePmChatStore.getState().finalizeResponse(requestId)
- pm-chat-error -> usePmChatStore.getState().setStreaming(false) + appendToken + finalizeResponse
- pm-chat-health -> usePmChatStore.getState().setHealth(ok, error)
- annotation-update -> useAnnotationBridgeStore.getState().setAnnotations(annotations)
- walkthrough-step -> useAnnotationBridgeStore.getState().setWalkthroughStep(step)
- plan-update -> usePlanStore.getState().setContent(content, fileName)
- diff-result -> useDiffStore.getState().setDiffs(parsed, raw)
- ask-code-response -> document.dispatchEvent(CustomEvent)
- mode-status -> useModeStore.getState().handleModeStatus(msg)
- mode-crash-recovery -> useModeStore.getState().handleCrashRecovery()
- default -> handleHistoryMessageRef.current(msg)

```typescript
import type { ServerMessage, AgentEvent, Annotation, WindowThumbnail } from '../protocol';

export interface DispatchCallbacks {
  write: (data: string) => void;
  setCurrentShell: (shell: string) => void;
  setShells: (shells: string[]) => void;
  setPendingImagePath: (path: string) => void;
  setPickerWindows: (windows: WindowThumbnail[]) => void;
  setPendingInjection: (block: string) => void;
  pushAgentEvent: (event: AgentEvent) => void;
  appendPmChatToken: (requestId: string, token: string) => void;
  finalizePmChatResponse: (requestId: string) => void;
  setPmChatStreaming: (streaming: boolean) => void;
  setPmChatHealth: (ok: boolean, error?: string) => void;
  setAnnotations: (annotations: Annotation[]) => void;
  setWalkthroughStep: (step: any) => void;
  setPlanContent: (content: string | null, fileName: string | null) => void;
  setDiffs: (raw: string) => void;
  dispatchAskCodeResponse: (msg: ServerMessage) => void;
  handleModeStatus: (msg: ServerMessage) => void;
  handleCrashRecovery: () => void;
  handleHistoryMessage: (msg: ServerMessage) => void;
  // Context for pty-exit:
  currentShell: string | null;
  paneId: string;
  shells: string[];
  recordCompleted: () => void;
  completionStatsEnabled: boolean;
  notifyExit: (info: { exitCode: number; shell: string; paneId: string }) => void;
  scheduleRespawn: () => void;
}
```

Export `resolveShellName(fullPath: string, shells: string[]): string` — pure function. Returns the shell from the shells list whose name matches the end of fullPath (case-insensitive), or fullPath if no match.

Export `dispatchServerMessage(msg: ServerMessage, cb: DispatchCallbacks): void` — pure function. Implements the 16+ case switch from TerminalPane lines 102-220, calling the appropriate callback for each message type. For `capture-result-with-metadata`, call `cb.setPendingInjection(formatCaptureBlock({...}))` — import `formatCaptureBlock` from `../utils/formatCaptureBlock`. For `diff-result`, call `cb.setDiffs(msg.raw)` (let the callback handle parsing). For `pty-ready`, use `resolveShellName(msg.shell, cb.shells)` internally.
