Create `e2e/smoke-pty-flow.spec.ts` — the core PTY flow E2E smoke test.

**Prerequisites** (same as uat-phase35 per D-13): Running Vite dev server (localhost:1420) + running sidecar (port from discovery file). No Ollama dependency.

**Reuse from uat-phase35.spec.ts:** Copy `getSidecarPort()` and `mockTauriAndConnect()` functions (or import if extracting to a shared helper is cleaner — use judgment per Claude's Discretion).

**Test structure:**

```typescript
import { test, expect, type Page } from '@playwright/test';
import * as fs from 'fs';
import * as path from 'path';

// -- getSidecarPort() and mockTauriAndConnect() (reuse from uat-phase35) --

test.describe('Smoke: Core PTY Flow', () => {
  let sidecarPort: number;

  test.beforeAll(() => {
    const discovery = getSidecarPort();
    sidecarPort = discovery.port;
  });

  test('TEST-01: CDP connection reads visible DOM element', async ({ page }) => {
    // This test validates Playwright can connect and interact with the app UI
    await mockTauriAndConnect(page, sidecarPort);
    await page.goto('http://localhost:1420');
    await page.waitForTimeout(1000);

    // Read a visible DOM element — the app header or terminal container
    // The AppHeader component always renders the app title area
    const body = page.locator('body');
    await expect(body).toBeVisible();

    // Verify we can read actual rendered content (not just an empty page)
    const html = await page.content();
    expect(html).toContain('</div>'); // App rendered React components
    
    // Check for a known UI element (ChatInputBar placeholder or header)
    const hasContent = await page.evaluate(() => {
      return document.querySelectorAll('[class]').length > 5;
    });
    expect(hasContent).toBe(true);
  });

  test('TEST-03: PTY flow — spawn, send command, verify output', async ({ page }) => {
    // Capture console for debugging
    const consoleLogs: string[] = [];
    page.on('console', (msg) => consoleLogs.push(`[${msg.type()}] ${msg.text()}`));

    await mockTauriAndConnect(page, sidecarPort);

    // Inject WebSocket message interceptor BEFORE page loads (D-10)
    await page.addInitScript((port) => {
      // Create test state accumulator on window
      (window as any).__testState = {
        wsConnected: false,
        ptyReady: false,
        outputBuffer: '',
        shellList: [] as string[],
        messages: [] as any[],
      };

      // Monkey-patch WebSocket to intercept messages
      const OrigWebSocket = window.WebSocket;
      (window as any).WebSocket = function(url: string, protocols?: string | string[]) {
        const ws = new OrigWebSocket(url, protocols);
        const state = (window as any).__testState;

        ws.addEventListener('open', () => {
          state.wsConnected = true;
        });

        ws.addEventListener('message', (event: MessageEvent) => {
          try {
            const msg = JSON.parse(event.data);
            state.messages.push(msg);

            if (msg.type === 'pty-ready') {
              state.ptyReady = true;
            }
            if (msg.type === 'output') {
              state.outputBuffer += msg.data;
            }
            if (msg.type === 'shell-list') {
              state.shellList = msg.shells;
            }
          } catch {}
        });

        return ws;
      } as any;
      // Copy static properties
      Object.assign((window as any).WebSocket, OrigWebSocket);
    }, sidecarPort);

    await page.goto('http://localhost:1420');

    // Step 1: Wait for WebSocket to connect (D-15)
    await page.waitForFunction(
      () => (window as any).__testState?.wsConnected === true,
      { timeout: 5000 }
    );

    // Step 2: Wait for pty-ready (ConPTY + PowerShell init takes 200-800ms per D-15)
    await page.waitForFunction(
      () => (window as any).__testState?.ptyReady === true,
      { timeout: 10000 }
    );

    // Step 3: Clear output buffer after PTY startup noise
    await page.evaluate(() => {
      (window as any).__testState.outputBuffer = '';
    });

    // Step 4: Send a command via the app's WebSocket (D-10)
    // Find the WebSocket instance and send an input message
    await page.evaluate(() => {
      // The app's WS sends JSON messages — find it by looking for the
      // active connection, or just use the input mechanism
      // Use the ChatInputBar's onSend path: type into textarea + Enter
      const textarea = document.querySelector('textarea');
      if (textarea) {
        // Simulate React-compatible input
        const nativeInputValueSetter = Object.getOwnPropertyDescriptor(
          window.HTMLTextAreaElement.prototype, 'value'
        )?.set;
        nativeInputValueSetter?.call(textarea, 'echo __PLAYWRIGHT_SMOKE__');
        textarea.dispatchEvent(new Event('input', { bubbles: true }));
        textarea.dispatchEvent(new Event('change', { bubbles: true }));
        // Press Enter
        textarea.dispatchEvent(new KeyboardEvent('keydown', {
          key: 'Enter', code: 'Enter', keyCode: 13, bubbles: true
        }));
      }
    });

    // Step 5: Accumulate output over 5-second window (D-12)
    // ConPTY echoes input AND output — buffer will contain both
    await page.waitForFunction(
      () => (window as any).__testState?.outputBuffer?.includes('__PLAYWRIGHT_SMOKE__'),
      { timeout: 10000 }
    );

    // Verify output buffer contains our marker
    const outputBuffer = await page.evaluate(
      () => (window as any).__testState?.outputBuffer
    );
    expect(outputBuffer).toContain('__PLAYWRIGHT_SMOKE__');

    // Log diagnostics
    const messageCount = await page.evaluate(
      () => (window as any).__testState?.messages?.length
    );
    console.log(`Smoke test complete: ${messageCount} WS messages captured`);
    console.log(`Output buffer length: ${outputBuffer?.length}`);
  });
});
```

**Key design choices per decisions:**
- Uses mock-Tauri + Vite dev server approach as primary (D-04, D-11) — most reliable without manual Tauri launch
- WebSocket interceptor via `addInitScript` monkey-patches WebSocket constructor (D-10)
- Accumulates all `output` messages into buffer, asserts substring match (D-12) — handles ConPTY chunking and echo
- Uses `__PLAYWRIGHT_SMOKE__` as a unique marker that won't appear in startup noise
- 10-second timeout for pty-ready (ConPTY + PowerShell init) per D-15
- 10-second timeout for output accumulation per D-12
- Falls back cleanly — if WebSocket doesn't connect, waitForFunction throws with clear timeout message
